# auto_trader/utils.py
import pandas as pd
from datetime import datetime, timezone

def utc_now():
    """현재 시각을 UTC 타임스탬프로 반환"""
    return pd.Timestamp.utcnow()

def interpolate_missing(df: pd.DataFrame) -> pd.DataFrame:
    """
    결측치(NaN)를 선형 보간으로 채워 반환.
    datetime 인덱스를 유지하면서 numeric 컬럼만 보간한다.
    """
    # 날짜/시간 컬럼은 그대로 두고, 수치형 컬럼만 보간
    numeric_cols = df.select_dtypes(include=['number']).columns
    df[numeric_cols] = df[numeric_cols].interpolate(method='linear')
    return df

def safe_division(numerator, denominator):
    """0DivisionError 방지를 위한 안전除法"""
    try:
        return numerator / denominator
    except ZeroDivisionError:
        return float('inf')

# auto_trader/broker.py
import requests
import json
from datetime import datetime, timezone

class BrokerAPI:
    """
    가상 브로커 API 래퍼.
    실제 거래소(예: Binance, Bybit 등)와 연동될 때는
    여기서 HTTP 요청만 교체하면 된다.
    """

    def __init__(self, base_url: str = "https://api.example.com"):
        self.base_url = base_url
        self.session = requests.Session()

    # ------------------------------------------------------------------
    # 계좌 조회
    # ------------------------------------------------------------------
    def get_account_balance(self) -> float:
        """
        계좌 평가를 반환 (USD 기준).
        실제 구현에서는 API 인증과 JSON 파싱을 수행한다.
        """
        resp = self.session.get(f"{self.base_url}/account")
        data = resp.json()
        return float(data["balance"])

    # ------------------------------------------------------------------
    # 포지션 조회
    # ------------------------------------------------------------------
    def get_position(self, symbol: str) -> dict:
        """
        특정 심볼에 대한 포지션 정보를 반환.
        예시 구조: {"symbol": "BTCUSDT", "size": 0.015, "side": "long"}
        """
        resp = self.session.get(f"{self.base_url}/position", params={"symbol": symbol})
        data = resp.json()
        return data

    # ------------------------------------------------------------------
    # 주문 전송
    # ------------------------------------------------------------------
    def place_order(self, symbol: str, qty: float, side: str) -> bool:
        """
        매수/매도 주문을 전송한다.
        성공 시 True, 실패 시 False 반환.
        """
        payload = {
            "symbol": symbol,
            "quantity": qty,
            "side": side,
            "type": "market",   # 시장가 order (예시)
            "time_in_force": "GTC"
        }
        resp = self.session.post(f"{self.base_url}/order", json=payload)
        result = resp.json()
        return result.get("status") == "success"

    # ------------------------------------------------------------------
    # 헬퍼: 포지션당 리스크 금액 계산
    # ------------------------------------------------------------------
    def calculate_position_size(self, equity: float, risk_percent: float, stop_loss_distance: float) -> float:
        """
        위험 금액 = equity * risk_percent
        포지션 사이즈 = 위험 금액 / stop_loss_distance
        """
        risk_amount = equity * risk_percent
        return safe_division(risk_amount, stop_loss_distance)

# auto_trader/strategy.py
import pandas as pd
from datetime import timezone

class EMARSI:
    """
    EMA-RSI 하이브리드 전략 클래스.
    - EMA 크로스(Fast > Slow) + RSI >= 50 이면 롱 진입
    - EMA 크로스( Fast < Slow) + RSI <= 50 이면 숏 진입
    - RSI 과매수(>70) 혹은 과매도(<30)에서 청산 알림
    """

    def __init__(self, fast_period: int = 9, slow_period: int = 21,
                 rsi_period: int = 14):
        self.fast_period = fast_period
        self.slow_period = slow_period
        self.rsi_period = rsi_period

    # --------------------------------------------------------------
    # 지표 계산 (DataFrame에 NaN 보간 전 호출)
    # --------------------------------------------------------------
    def compute_indicators(self, df: pd.DataFrame) -> pd.DataFrame:
        """
        EMA와 RSI를 추가하고 'signal' 컬럼을 만든다.
        signal:  1 = 롱, -1 = 숏, 0 = 홀드
        """
        df = df.copy()

        # EMA
        df["ema_fast"] = df["close"].ewm(span=self.fast_period, adjust=False).mean()
        df["ema_slow"] = df["close"].ewm(span=self.slow_period, adjust=False).mean()

        # RSI
        delta = df["close"].diff()
        up = delta.where(delta > 0, 0)
        down = delta.where(delta < 0, 0)

        roll_up = up.ewm(alpha=1 / self.rsi_period, adjust=False).mean()
        roll_down = down.ewm(alpha=1 / self.rsi_period, adjust=False).mean()

        rs = roll_up / roll_down
        df["rsi"] = 100 - (100 / (1 + rs))

        # 시그널 생성
        df["signal"] = 0
        # EMA 크로스 업
        df.loc[(df["ema_fast"] > df["ema_slow"]) & (df["rsi"] >= 50), "signal"] = 1   # 롱
        # EMA 크로스 다운
        df.loc[(df["ema_fast"] < df["ema_slow"]) & (df["rsi"] <= 50), "signal"] = -1  # 숏

        return df

    # --------------------------------------------------------------
    # 현재 시그널 결정 (마지막 행 기준)
    # --------------------------------------------------------------
    def decide_trade(self, df: pd.DataFrame) -> dict:
        """
        최신 데이터를 기반으로 trade 결정을 반환.
        {
            "action": "buy" | "sell" | "hold",
            "signal": 1 | -1 | 0,
            "reason": str
        }
        """
        if df.empty:
            return {"action": "hold", "signal": 0, "reason": "no data"}

        latest = df.iloc[-1]
        prev = df.iloc[-2] if len(df) > 1 else None

        # EMA 크로스 확인
        ema_cross_up = (prev["ema_fast"] <= prev["ema_slow"]) and (latest["ema_fast"] > latest["ema_slow"])
        ema_cross_down = (prev["ema_fast"] >= prev["ema_slow"]) and (latest["ema_fast"] < latest["ema_slow"])

        # RSI 조건
        rsi_ok_long = latest["rsi"] >= 50
        rsi_ok_short = latest["rsi"] <= 50

        # 청산 조건 (과매수/과매도)
        if latest["rsi"] > 70:
            return {"action": "sell", "signal": -1, "reason": "RSI overbought"}
        if latest["rsi"] < 30:
            return {"action": "buy", "signal": 1, "reason": "RSI oversold"}

        # 진입 판단
        if ema_cross_up and rsi_ok_long:
            return {"action": "buy", "signal": 1, "reason": "EMA bullish crossover + RSI OK"}
        if ema_cross_down and rsi_ok_short:
            return {"action": "sell", "signal": -1, "reason": "EMA bearish crossover + RSI OK"}

        return {"action": "hold", "signal": 0, "reason": "no clear signal"}

# auto_trader/backtest.py
import pandas as pd
from datetime import timedelta

class Backtester:
    """
    간단한 백테스팅 엔진.
    - 일일 손실 한도(3%)를 초과하면 자동 중단.
    - 포지션당 리스크(1%)를 적용해 포지션 사이즈 계산.
    """

    def __init__(self, config):
        self.equity = config.get("initial_equity", 10000.0)   # 시작 капитал
        self.daily_loss_limit_pct = config["daily_loss_limit"]
        self.risk_percent = config["risk_percent"]
        self.stop_loss_atr_mult = config.get("stop_loss_atr_mult", 1.5)
        self.trades = []      # 리스트에 (timestamp, action, size, pnl) 저장

    def run(self, df: pd.DataFrame):
        """
        df must contain columns: ['open','high','low','close','volume'] and a datetime index.
        """
        df = df.copy()
        df["ema_fast"] = df["close"].ewm(span=self.config["fast_period"], adjust=False).mean()
        df["ema_slow"] = df["close"].ewm(span=self.config["slow_period"], adjust=False).mean()

        position = 0   # 1=롱, -1=숏, 0=flat
        entry_price = 0.0
        entry_time = None

        for i, row in df.iterrows():
            # 지표 계산 (EMA, RSI) – 여기서는 EMA만 사용
            if position != 0:
                # 간단하게 EMA 크로스에서_exit
                if (position == 1 and row["ema_fast"] < row["ema_slow"]) or \
                   (position == -1 and row["ema_fast"] > row["ema_slow"]):
                    pnl = (row["close"] - entry_price) * position if position == 1 else (entry_price - row["close"]) * position
                    self.trades.append((i, "close", position, pnl))
                    self.equity += pnl
                    position = 0
                    entry_price = 0.0

            # 신규 진입
            signal = 1 if (row["ema_fast"] > row["ema_slow"]) else -1 if (row["ema_fast"] < row["ema_slow"]) else 0
            if signal != 0:
                # 손실 허용 거리 (ATR 대신 EMA 차이로 간단히 계산)
                atr = abs(row["ema_fast"] - row["ema_slow"])
                stop_loss_dist = atr * self.stop_loss_atr_mult
                risk_amount = self.equity * self.risk_percent
                qty = self.config["broker"].calculate_position_size(self.equity, self.risk_percent, stop_loss_dist)

                # 실제 주문 시뮬레이션 (가격은 현재 종가)
                entry_price = row["close"]
                position = signal
                entry_time = i
                self.trades.append((i, "open", qty, 0.0))   # pnl은 나중에 계산

        # 일일 손실 한도 체크
        daily_loss = 0.0
        for t, _, _, pnl in self.trades:
            daily_loss += pnl
            if abs(daily_loss) > self.equity * self.daily_loss_limit_pct:
                print(f"[BACKTEST] 일일 손실 한도 초과 ({daily_loss:.2f}%): 테스트 중단")
                break

        return self.trades, self.equity

### 🛠️ [통합 아키텍처의 핵심 가치 및 검증]

이 구조는 다음 세 가지 측면에서 기존 방식보다 압도적으로 우수합니다:

1.  **강력한 결합도 분리 (Decoupling):** `DataStreamManager`가 망가지든, `TradingEngine`의 로직이 바뀌든, `StateManagerAPI`는 오직 **이벤트(Event)**만 받아서 상태를 업데이트하면 됩니다. 각 모듈은 독립적으로 테스트 가능합니다.
2.  **실시간성 보장 (Real-time):** 모든 최종 결정과 KPI 변화가 `StateManagerAPI`를 통해 중앙 집중화되고, 이는 WebSocket을 통해 클라이언트에 푸시됩니다. 백테스팅 결과(Batch)와 실시간 운영(Streaming) 데이터가 같은 '상태 관리층'을 공유하게 되어 **데이터 불일치 문제가 원천적으로 차단**됩니다.
3.  **리스크 우선주의 (Risk-First):** 신호가 발생하면 $\rightarrow$ `TradingEngine` $\rightarrow$ `StateManagerAPI`에 도달하기 전에, 반드시 `RiskManagerService`를 거칩니다. 이는 회사의 최우선 가치인 **'자본 보존' 로직을 코어 흐름의 가장 앞단에 배치**했음을 의미합니다.

### 📈 Premium Monitoring Service (PMS) 가치 재평가 및 판매 전략 수정안

#### 1. 핵심 컨셉 변경: '신호 제공' $\rightarrow$ '리스크 보험'
기존의 PMS는 단순히 "좋은 매매 신호를 알려주는 서비스"였다면, 기술적 안정성 확보 후에는 **"어떤 시장 충격(Black Swan)에도 자본을 보전하고 복구하는 리스크 관리 솔루션"**으로 포지셔닝해야 합니다.

| 영역 | 이전 컨셉 (신호 기반) | 변경된 컨셉 (리스크/안정성 기반) | 판매 강점 |
| :--- | :--- | :--- | :--- |
| **핵심 가치** | 높은 수익률 달성 기회 제공 | 자본 보존 및 손실 최소화 확신 제공 | 신뢰도와 생존 가능성을 파는 것. |
| **고객 Pain Point** | "돈을 잃을까 봐 불안하다." | "위기 상황에서 대응할 능력이 없다." | 공포(Fear)를 자극하고, 해결책으로 보험을 제시. |
| **판매 포인트** | 높은 ROI 예상 (A안/B안 비교) | **시스템의 회복력 및 검증된 리스크 지표.** | 백테스팅 데이터가 아닌, '검증 프로세스' 자체 판매. |

#### 2. 시장에 팔 수 있는 핵심 KPI (KPIs for Saleability)
기술적인 용어(Mockup, End-to-End Integration 등)를 고객이 이해하고 돈을 지불할 만한 언어로 변환해야 합니다. PMS의 가격 책정 근거가 될 구체적이고 계량화된 지표들입니다.

1.  **최대 낙폭 대비 회복률 (Max Drawdown Recovery Rate, MDRR):**
    *   *정의:* 시스템이 최대로 손실을 본 시점(Drawdown)에서 일정 목표치까지 복구하는 데 걸린 시간과 그 과정을 리스크 지표로 제시합니다.
    *   *판매 효과:* "최대 위험에 빠졌더라도, 우리 시스템은 평균 $X$일 내에 자본 보존 라인으로 회복했습니다." (회복 속도 판매)
2.  **스트레스 테스트 성공 비율 (Stress Test Success Ratio, STSR):**
    *   *정의:* 롤러코스터급 변동성(예: 급격한 금리 인상 시나리오, 지정학적 이벤트 시뮬레이션 등)과 같은 '위기 상황'을 백테스팅에 강제 주입했을 때, 리스크 규칙 위반 없이 살아남은 테스트 케이스의 비율.
    *   *판매 효과:* "최근 5년간 발생한 모든 시장 충격 중 $X$개의 시나리오를 성공적으로 통과했습니다." (검증된 경험 판매)
3.  **포지션 자율성 지수 (Position Autonomy Index, PAI):**
    *   *정의:* 시스템이 인간의 감정(공포/탐욕에 의한 과매수/과매도 매매) 개입 없이 오직 설정된 리스크 규칙만을 따르며 거래를 수행할 수 있음을 증명하는 지표.
    *   *판매 효과:* "감정에 휘둘리는 트레이딩을 배제하고, 100% 기계적이고 객관적인 원칙에 기반합니다." (객관성/신뢰도 판매)

#### 3. 서비스 성공 기준 (Success Criteria for MVP V2.0)
PMS가 시장에서 '프리미엄'으로 인정받기 위한 최소한의 기술적, 보고서적 완성도를 정의했습니다. 이 세 가지를 충족해야 다음 단계로 진행 가능합니다.

1.  **기술적 안정성:** 백테스팅 결과가 단순 수익률(Total Return) 외에 **MDRR, STSR 등 리스크 지표를 산출하고 이를 시각화하는 기능**이 구현되어야 한다. (개발팀/Developer 담당)
2.  **보고서 구조 확정:** 이 수치들을 기반으로 'Premium Monitoring Service'의 초기 보고서 템플릿을 완성한다. 단순 차트가 아닌, **리스크 관리자(Risk Manager)가 만족할 만한 상세 설명과 해석이 포함되어야 한다.** (현빈 담당)
3.  **최초 가설 검증:** 이 KPI를 가장 잘 활용할 수 있는 초기 잠재 고객 그룹(예: 소액 투자처의 자금 규모가 큰 기관/개인 등)을 선정하고, **Pain Point에 맞는 맞춤형 메시지 1개를 작성하여 테스트한다.** (현빈 담당)

# 🛡️ Premium Monitoring Service: 시장이 요구하는 새로운 기준, '리스크 보험(Risk Insurance)'

## Executive Summary (경영진 요약)

**[문제 정의]** 기존의 자동 트레이딩 시스템들은 평상시의 *최적 수익률*만 보여주도록 설계되었습니다. 하지만 시장은 항상 예측 불가능한 블랙 스완(Black Swan) 이벤트와 극심한 변동성 구간으로 가득합니다. 현존하는 솔루션들은 위기 상황에서 시스템이 **'어떻게 붕괴하지 않고 버티는가'**에 대한 근본적인 질문에 답할 수 없습니다.

**[우리의 제안]** Premium Monitoring Service는 단순히 수익을 추적하는 대시보드가 아닙니다. 이는 극한의 스트레스 환경(Stress Test)을 통과하도록 설계된 **‘자본 보존 시스템’**입니다. 우리는 트레이딩 성과가 아닌, **'회복력(Resilience)'**과 **'안전성'**에 대한 KPI를 판매합니다.

## 🎯 왜 기존 솔루션은 위험한가? (Pain Point & Gap Analysis)

대부분의 경쟁사들이 제시하는 지표는 다음과 같은 한계점을 가집니다:
*   **평균 수익률 기반:** 시장이 좋을 때만 높은 숫자를 보여줍니다.
*   **최대 낙폭(Max Drawdown) 누락:** 손실을 겪은 후 얼마나 회복할지, 그 과정의 안정성을 무시합니다.
*   **예외 상황 처리 미반영:** 네트워크 장애, API 연결 끊김 등 '시스템 실패' 시나리오를 테스트하지 않습니다.

우리는 이 간극(Gap)을 메워, **‘만일의 사태’에 대한 확신**을 제공합니다.

    timestamp: str  # UTC 기준 ISO 8601 형식
    current_drawdown: float # 현재 자본 대비 손실률 (%)
    rsi: float            # 시스템 복원력 지수 (Resilience Index)
    mdrr: float           # 최대 낙폭 대비 회복률 (Recovery Rate, %)
    system_status: str    # System Status (Normal, Warning, Critical)

class StressTestResult(BaseModel):
    """전체 스트레스 테스트 결과 리스트."""
    metrics: List[StressMetric]
    summary: Dict[str, float] # 최종 요약 지표

def generate_mock_data(num_points: int = 10) -> List[StressMetric]:
    """
    가상의 스트레스 테스트 데이터 포인트를 생성합니다.
    시뮬레이션 로직: 초기 안정 상태 -> 갑작스러운 하락 (스트레스) -> 점진적 회복 시도.
    """
    metrics = []
    start_time = datetime.utcnow() - timedelta(minutes=num_points * 5)

    # 초기값 설정 (안정적인 상태로 시작)
    current_drawdown = 0.1
    rsi = 90.0 # 높음 = 안정적
    mdrr = 95.0 # 높음 = 회복력 좋음

    for i in range(num_points):
        timestamp = (start_time + timedelta(minutes=i * 5)).isoformat() + 'Z'

        # 1. 스트레스 주입 로직 (가장 변동성이 큰 시점에 Drawdown과 RSI 하락)
        if i > num_points // 2 and i < num_points - 2:
            # 시장 급변성 구간 -> 자본 손실 증가, 복원력 지수 하락
            current_drawdown += random.uniform(0.5, 1.5) * (i / num_points)
            rsi = np.clip(rsi - random.uniform(3, 7), 20, 90) # 급격히 떨어짐
            mdrr -= random.uniform(1, 3) # 회복력 저하
        else:
            # 안정 또는 회복 구간 -> 손실은 유지되거나 감소하며 복원력을 조금씩 끌어올림
            current_drawdown = max(0.1, current_drawdown + random.uniform(-0.2, 0.3))
            rsi = np.clip(rsi + random.uniform(0.5, 2), 40, 90) # 서서히 회복
            mdrr = min(100.0, mdrr + random.uniform(0.5, 2))

        # 2. 안전장치 및 클리핑 (최대/최소 범위 제한)
        current_drawdown = np.clip(current_drawdown, 0.0, 10.0) # 최대 낙폭 10%로 제한
        rsi = np.clip(rsi, 20, 95)
        mdrr = np.clip(mdrr, 50, 100)

        # 3. Status 결정 (리스크 지표 기반)
        if current_drawdown > 5.0 or rsi < 40:
            status = "Critical" # 위험 임계치 초과
        elif current_drawdown > 2.5 or rsi < 60:
            status = "Warning" # 경고 단계 진입
        else:
            status = "Normal"

        metrics.append(StressMetric(
            timestamp=timestamp,
            current_drawdown=round(current_drawdown, 2),
            rsi=round(rsi, 1),
            mdrr=round(mdrr, 1),
            system_status=status
        ))

    # 요약 지표 계산 (최종 drawdnow, 평균 RSI 등)
    summary = {
        "final_drawdown": round(metrics[-1].current_drawdown, 2),
        "average_rsi": round(np.mean([m.rsi for m in metrics]), 1),
        "min_mdrr": round(min([m.mdrr for m in metrics]), 1)
    }

    return StressTestResult(metrics=metrics, summary=summary)

# 📑 Premium Monitoring Service: MVP 핵심 KPI 스펙 시트 (Pro Guard & Enterprise Elite Target)

## 🎯 목표 설정 배경 (The 'Why')
우리의 타깃 청중은 돈을 벌고 싶지만, 그보다 **'넣은 돈을 지키고 싶어 하는 사람'**입니다. 기존 대시보드가 $P&L$ 곡선(수익 흐름)에 집중했다면, 우리의 대시보드는 $Risk$ 커브(위험 방지력)에 초점을 맞춰야 합니다. 아래 KPI들은 모두 '최악의 시나리오에서도 살아남았음'을 증명합니다.

## 🔑 핵심 지표 상세 명세 (The 'What')

### 🥇 KPI 1: 최대 낙폭 회복률 (Maximum Drawdown Recovery Rate, MDRR)
**[가장 매력적인 판매 포인트]**
최대 낙폭(Max Drawdown)은 시스템이 최악의 위기를 맞았을 때까지의 하락 폭입니다. 이 지표는 "우리가 얼마나 빨리 털고 일어나는가"를 보여주며, 트레이딩 실패에 대한 가장 직관적인 보험 개념을 제공합니다.

*   **수학적 정의 (Formula):**
    $$MDRR = \frac{\text{Current Equity} - \text{Max Drawdown}}{\text{Max Drawdown}} \times 100\%$$
    *(참고: MDRR은 초기 자본 대비 회복률로 계산되어, 값이 높을수록 위험 상황 발생 후 빠르게 정상화되었음을 의미합니다.)*

*   **데이터 소스 (Data Source):**
    1.  **필수 데이터:** 백테스팅 기간 동안의 시간별/일별 **순자산 가치(Equity Curve)** 시계열 데이터.
    2.  **추가 요구 사항:** 가장 깊었던 낙폭 지점($\text{Max Drawdown}$)을 정확히 포착할 수 있는 롤백(Rollback) 기능 및 로그 기록.

### 🥈 KPI 2: 자본 보존 비율 (Capital Preservation Ratio, CPR)
**[Pro Guard 핵심 지표]**
이것은 단순히 손실률을 넘어, '사전에 정의된 안전장치'가 얼마나 효과적이었는지를 보여줍니다. 트레이더가 설정한 최대 허용 리스크(예: 5% 이하로 떨어지면 자동 매매 중단)를 시스템이 성공적으로 지켰는지 증명합니다.

*   **수학적 정의 (Formula):**
    $$CPR = \frac{\text{Minimum Observed Equity} / \text{Initial Capital}}{\text{Pre-defined Safety Threshold}}$$
    *(참고: CPR은 1에 가까울수록 초기 자본을 잘 지켰음을 의미하며, '안전 마진' 관점에서 해석됩니다.)*

*   **데이터 소스 (Data Source):**
    1.  **필수 데이터:** 시스템이 작동하는 동안의 **계좌별 거래 내역 및 포지션 크기(Position Size)** 로그.
    2.  **추가 요구 사항:** 사용자가 설정한 '리스크 한도' 값(Threshold)을 기준으로, 해당 리스크 위반 발생 시점과 그 성공적인 차단 기록이 필수적으로 로깅되어야 합니다.

### 🥉 KPI 3: 시스템 복원력 지수 (System Resilience Index, SRI)
**[Enterprise Elite 독점 지표]**
이것은 가장 고도화된 지표로, 트레이딩 전략 자체가 아닌 **'시스템의 운영 안정성'을 판매합니다.** API 연결 끊김, 데이터 결측치 발생(Data Gap), 서버 장애 등 외부 요인으로 인해 시스템에 문제가 생겼을 때, 얼마나 빠르게 자가 진단하고 정상 작동 상태로 복귀했는지를 점수화한 지표입니다.

*   **수학적 정의 (Formula):**
    $$SRI = \sum_{t=1}^{T} w_i \cdot (\text{Operational\_Check}_t)$$
    *(여기서 $w_i$는 각 체크 항목의 중요 가중치, $\text{Operational\_Check}$는 데이터 안정성(Data Stability), API 연결성(Connectivity), 로직 무결성(Integrity) 등의 성공 여부를 나타내는 지표입니다. 100점에 가까울수록 완벽함을 의미합니다.)*

*   **데이터 소스 (Data Source):**
    1.  **필수 데이터:** 모든 외부 연동 시스템(API Gateway, Data Feed)의 **운영 로그(Operational Logs)** 및 에러 트랜잭션 기록.
    2.  **추가 요구 사항:** 단순 오류 발생 카운트가 아니라, '오류 감지 $\rightarrow$ 경고 발령 $\rightarrow$ 대체 로직 실행 $\rightarrow$ 복구 완료'에 이르는 **전체 프로세스 성공 시간(Mean Time To Recovery, MTTR)**을 측정해야 합니다.

*(Subtitle)* 당신의 시스템은 시장의 블랙 스완(Black Swan)에 대비되어 있습니까?

*   **핵심 메시지:** 기존 솔루션들은 '평균적인 좋은 날'만 보여줍니다. 하지만 트레이딩에서 가장 중요한 것은 **가장 나쁜 날을 버티는 능력**입니다.
*   **시각화 요청:** 폭락하는 시장 그래프 (빨간색) 위에, 안정적으로 자본을 유지하며 움직이는 안전한 방패 아이콘(Blue/Green) 오버레이.

## Slide 2: 경쟁사의 한계와 우리의 가치 제안 (The Gap Analysis)
**[제목]** **리스크 지표의 부재가 가장 큰 위험입니다.**
*(Subtitle)* 저희는 수익률을 팔지 않습니다. 우리는 '시스템 생존력'을 판매합니다.

*   **경쟁사 솔루션:** PnL(Profit & Loss) 및 일일/주간 수익 그래프 중심. (→ "오늘 얼마나 벌었나요?")
    *   *(문제점 지적):* 급격한 변동성 발생 시, 최대 손실 구간에 대한 경고나 회복력 지표가 없습니다.
*   **✨ 우리의 솔루션:** 시스템 복원력(Resilience) 중심의 KPI 제공. (→ **"최악의 상황에서도 어떻게 버틸 수 있나요?"**)
    *   **핵심 키워드 강조:** 자본 보존, 스트레스 테스트, 예외 처리 로직 검증.

## Slide 3: 핵심 증거 제시 - '안전장치' KPI 소개 (The Proof - Data Focus)
**[제목]** **독보적인 리스크 관리 지표(KPI): 시장의 불안함을 수치화하다.**
*(Subtitle)* Pro Guard와 Enterprise Elite 등급에서만 제공하는, 생존을 위한 3가지 핵심 지표.

| KPI | 정의 (고객 이해용 언어) | 왜 중요한가? (Pain Point 해소) | 시각화 요청 |
| :--- | :--- | :--- | :--- |
| **MDRR** *(Max Drawdown Recovery Rate)* | 최대 낙폭 대비 회복률. 가장 깊은 하락장에서 얼마나 빠르게 기준 자본금에 근접하는지를 측정합니다. | "최악의 순간, 시스템이 멈추지 않고 돌아올 힘을 가졌음을 증명합니다." (생존력) | 게이지 차트: 목표치 대비 현재값(Green/Yellow/Red) |
| **CPR** *(Capital Preservation Ratio)* | 자본 보존 비율. 포지션별 최대 손실 한도 위반 횟수 및 총액을 모니터링합니다. | "단순히 돈을 벌게 하는 것이 아니라, '돈을 지키는' 원칙을 강제합니다." (안전성) | 추이 그래프: 시간 경과에 따른 자본금 안전 마진 변화 |
| **시스템 복원력 지수** | API 연결 끊김, 데이터 결측치 등 예외 상황 발생 시 시스템의 정상화 속도와 안정성을 수치로 나타냅니다. | "예상치 못한 외부 충격(기술적 문제)에서도 트레이딩이 중단되지 않도록 보장합니다." (운영 신뢰성) | 흐름도/KPI: 장애 감지 시간 대비 복구 완료 시간 비율 |

## Slide 4: 가격 및 패키징 구조 (The Offer - Actionable Next Step)
**[제목]** **고객의 니즈에 맞는 최적의 '안전망'을 선택하세요.**
*(Subtitle)* Basic은 경고를, Elite는 완벽한 시스템 생존까지 설계했습니다.

*   **Basic Monitoring:**
    *   **제공 가치:** 실시간 PnL 및 일반적인 리스크 알림 (Drawdown Alert).

    *   **비용 대비 이점:** 저렴한 비용으로 최소한의 안정성을 확보합니다.

*   **Pro Guard Monitoring (⭐추천):**
    *   **제공 가치:** **MDRR 및 CPR 모니터링 추가.** 핵심 위험 지표 2가지 제공.
    *   **누구에게 필요한가:** 경험이 있는 트레이더, 시장 변동성에 노출되어 불안감을 느끼는 분.
    *   **핵심 메시지:** "불안한 시장에서 당신의 자본을 과학적으로 방어합니다."

*   **Enterprise Elite (최상위):**
    *   **제공 가치:** **Pro Guard + 시스템 복원력 지수(RSI) 및 전체 E2E 스트레스 테스트 리포트.** 모든 예외 상황 커버.
    *   **누구에게 필요한가:** 기관급 자금 운용사, 자본 보존이 생명인 고액 트레이더.
    *   **핵심 메시지:** "시장 충격으로부터 완벽하게 분리된, 살아있는 금융 보험입니다."
