/**
 * Storyboard의 현재 Act와 시간을 기반으로 시뮬레이션 데이터를 생성합니다.
 * @param {number} timeSeconds - 경과 시간 (초).

 * @returns {{ohlc: number[], rsi: number[]}} 시뮬레이션된 데이터.
 */
const generateSimulationData = (timeSeconds, currentAct) => {
    let volatility;

    switch (currentAct) {
        case 'ACT_I': // Panic - 고변동성, 예측 불가능한 급등락
            volatility = 1 + Math.sin(timeSeconds / 5) * 0.8 + Math.random() * 0.5;
            break;
        case 'ACT_II': // Relief - 점진적 안정화, 작은 파동 위주
            // Act I 대비 변동성을 급격히 낮춥니다. (자본 보존의 개념 시각화)
            volatility = Math.max(0.5, 1.5 - (timeSeconds / 60)); // 시간 경과에 따라 감소
            break;
        case 'ACT_III': // Certainty - 매우 안정적이고 예측 가능한 흐름
            volatility = 0.2 + Math.random() * 0.1;
            break;
        default:
            volatility = 1.0;
    }

    // 가상의 데이터 생성 (실제 트레이딩 로직을 여기에 통합해야 함)
    const simulatedData = Array(30).fill(null).map((_, i) => ({
        time: Math.floor(i / 5) * 5, // 5초 단위 데이터 포인트 가정
        open: Math.random() * volatility + 100,
        high: Math.random() * volatility + 102,
        low: Math.random() * volatility - 1,
        close: (Math.random() * volatility + 98).toFixed(2),
        rsi: Math.min(70, Math.max(30, 50 + Math.sin((i / 5) * 0.5) * 20)).toFixed(1) // RSI 시뮬레이션
    }));

    return { ohlc: simulatedData.map(d => ({ open: parseFloat(d.open.toFixed(2)), high: parseFloat(d.high.toFixed(2)), low: parseFloat(d.low.toFixed(2)), close: parseFloat(d.close) })), rsi: simulatedData.map(d => d.rsi) };
};

## III. 타이포그래피 시스템 (Hierarchy & Emphasis)
*   **폰트 패밀리:** [고딕 계열의 가독성 높은 서체]를 사용하되, 전문적이고 기술적인 느낌을 강조합니다. (예: Pretendard/Noto Sans KR - Weight 활용 극대화)
*   **제목(H1):** **굵게(Bold)** 처리하며, 슬라이드 전체에서 가장 큰 비중을 차지하는 메시지(Pain Point)를 간결하게 전달해야 합니다. 서사적 흐름에 맞춰 '공포 $\to$ 안도 $\to$ 확신'의 키워드를 색상으로 강조합니다.
*   **데이터/KPI:** 폰트 크기보다 **색상과 시각적 배치가 중요**합니다. 숫자 자체를 강조할 때는 오직 `#008080` (확신) 계열로만 포인트를 주고, 다른 곳에서는 중립적인 색상을 사용해 신뢰성을 높입니다.

## IV. 핵심 시각 요소별 가이드라인 (Visual Asset Blueprint)
### 1. 리스크 지표 시각화 (The Gauge/Chart)
*   **공포 구간:** 그래프 라인은 `#A82D2D`를 메인으로 사용하며, 변동성(Volatility)이 클수록 선의 두께가 과장되어 불안감을 조성합니다. 'Max Drawdown' 영역은 붉은색 박스로 강하게 경계 처리합니다.
*   **안도 구간:** 그래프에 작은 패턴(예: 점선 또는 오렌지색 마커)을 삽입하여 '시스템 개입 신호'를 시각화합니다. 이 패턴이 곧 솔루션의 존재 증거입니다.
*   **확신 구간:** 안정화된 라인과 함께, 과거의 급격한 변동성을 덮어버리는 듯한 부드러운 아웃라인 효과(Teal 계열)를 적용하여 '영구적인 안전지대'처럼 보이게 합니다.

### 2. 흐름도/프로세스 시각화 (The Narrative Flow Diagram)
*   **레이아웃:** 선형적 배열보다, **사건의 전개에 따른 시간축(Timeline)** 또는 **피라미드 형태(문제-해결-궁극 목표)**를 활용합니다.
*   **연출:** 단순한 화살표가 아닌, '위기 발생 $\to$ 시스템 감지 $\to$ 데이터 재계산 $\to$ 안정화'의 과정을 보여주는 역동적인 커브 라인을 사용해야 합니다. 이 커브는 Teal 계열로 마감되어야 합니다.

### 3. CTA 및 행동 촉구 (The Certainty Anchor)
*   **원칙:** CTA 버튼은 항상 `#008080` 코어 트러스트 Teal을 사용하여, 모든 의사 결정의 최종 도착지가 '안전하고 통제된 영역'임을 각인시킵니다.
*   **메시지 강화:** "더 많은 수익" 대신 **"자본 보존 설계가 가능합니다."** 라는 문구를 메인 헤드라인으로 사용해야 합니다.
